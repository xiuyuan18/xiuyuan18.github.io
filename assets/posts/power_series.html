import numpy as np

from scipy.integrate import solve_ivp
import matplotlib.pyplot as plt

init_printing()  # pretty printing for sympy

Solve the equation $$y''(x)+2(x-1)y'(x)+xy(x)=0$$

a0, a1, x = symbols("a0, a1, x")
a2 = Rational(-1, 2) * a0


def series(n):
    x_powers = map(lambda n: (x - 1) ** n, count(0))
    state = deque([a2, a1, a0], maxlen=3)
    if n <= 2:
        coffs = list(reversed(state))[: n + 1]
        return sum(c * x_p for c, x_p in zip(coffs, islice(x_powers, n + 1)))
    else:
        sumation = series(2)
        x_it = islice(x_powers, 3, n + 1)
        for i in range(1, n - 1):
            new_state = (
                Rational(-(2 * i + 1), (i + 2) * (i + 1)) * state[1]
                + Rational(-1, (i + 2) * (i + 1)) * state[2]
            )
            state.appendleft(new_state)
            sumation += new_state * next(x_it)
        return sumation

def func(t, y):
    A = np.array([[0, 1], [-t, -2 * (t - 1)]])
    return A @ y


y0 = [1, 2]
t0 = 1.0

# Solve forward
t_span_fwd = (t0, 3)
t_eval_fwd = np.linspace(*t_span_fwd, int((t_span_fwd[1] - t_span_fwd[0]) * 20) + 1)
sol_fwd = solve_ivp(func, t_span_fwd, y0, t_eval=t_eval_fwd, rtol=1e-10, atol=1e-10)

# Solve backward
t_span_bwd = (t0, -1)
t_eval_bwd = np.linspace(*t_span_bwd, int((t_span_bwd[0] - t_span_bwd[1]) * 20) + 1)
sol_bwd = solve_ivp(func, t_span_bwd, y0, t_eval=t_eval_bwd, rtol=1e-10, atol=1e-10)

# Combine results
t_combined = np.concatenate((np.flip(sol_bwd.t), sol_fwd.t[1:]))
y_combined = np.concatenate((np.flip(sol_bwd.y[0]), sol_fwd.y[0][1:]))

# Plot the results getting from both methods
plt.figure(figsize=(10, 6))
for order in [5, 10, 15]:
    series_approx = series(order).subs({a0: y0[0], a1: y0[1]})
    series_approx_vectorized = np.vectorize(lambda t: series_approx.subs({x: t}))
    plt.plot(
        t_combined,
        series_approx_vectorized(t_combined),
        "-",
        label=f"Power series (n={order})",
    )

plt.plot(t_combined, y_combined, "k--", linewidth=2, label="solve_ivp solution")
plt.xlabel("t")
plt.ylabel("y(t)")
plt.ylim(-5, 5)  # Control y-axis range
plt.legend()
plt.grid(True)
plt.show()
